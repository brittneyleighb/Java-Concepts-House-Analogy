// Class = Blueprint for creating houses public class House { // Fields = Characteristics (like windows) private String address; private int numberOfRooms; private String color; // Constructor = Main entrance (door) public House(String address, int rooms, String color) { this.address = address; this.numberOfRooms = rooms; this.color = color; } // Methods = Actions the house can do public String getAddress() { return address; } public void paintHouse(String newColor) { this.color = newColor; System.out.println("House painted " + newColor); } public void displayInfo() { System.out.println("Address: " + address); System.out.println("Rooms: " + numberOfRooms); System.out.println("Color: " + color); } } // Creating objects/instances from the class public class Main { public static void main(String[] args) { // Object = Actual house built from blueprint House myHouse = new House("123 Main St", 3, "Blue"); House neighborHouse = new House("125 Main St", 4, "Red"); myHouse.displayInfo(); myHouse.paintHouse("Green"); } }
